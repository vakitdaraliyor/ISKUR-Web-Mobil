[ServiceContract]
public interface IServiceBlog
{
	[OperationContract]
	public List<KATEGORI> GetAllKATEGORI();
}

[DataContract]
public class Kategori
{
	[DataMember]
	public int KATEGORI_REFNO;
	[DataMember]
	public string KATEGORI_ADI;
}

ServiceBlog.svc.cs dosyasında
public class ServiceBlog:IService
{
	Model1 db = new Model1();
	public List<KATEGORI> GetAllKATEGORI() // interface teki methodu implement ettik
	{
		var liste = db.KATEGORIs.ToList();
		var listegonder = new List<Kategori>(); // gönderilecek liste
		for(int i=0; i<liste.Count; i++)
		{
			Kategori k = new Kategori();
			k.KATEGORI_ADI = liste[i].KATEGORI_ADI;
			k.KATEGORI_REFNO = liste[i].KATEGORI_REFNO;
			listegonder.Add(k);
		}
		return listegonder;
	}
}
